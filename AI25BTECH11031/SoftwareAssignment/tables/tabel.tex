\section{Index}
\[
\text{Frobenius Error} = \|A - A_k\|_F = \sqrt{\sum_{i=1}^{m}\sum_{j=1}^{n} (A_{ij} - A_{k,ij})^2}
\]
\[
\text{Relative Error} = \frac{\|A - A_k\|_F}{\|A\|_F}
\]
\[
\text{Compression Ratio} = \frac{m \times n}{k (m + n + 1)}
\]
